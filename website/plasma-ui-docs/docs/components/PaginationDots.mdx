## Использование

### Комплексный компонент – SmartPaginationDots
```tsx live
import React from 'react';

import { SmartPaginationDots } from '@sberdevices/plasma-ui';

export function App() {
    const index = 0;
    const items = [0, 0, 0, 0, 0, 0, 0, 0];

    return (
        <SmartPaginationDots items={items} index={index} visibleItems={7} />
    );
}
```

### Пользовательская верстка
```tsx live
import React from 'react';
import { PaginationDots, PaginationDot } from '@sberdevices/plasma-ui';

export function App() {
    const index = 0;
    const items = [0, 0, 0, 0, 0, 0, 0, 0];

    return (
        <PaginationDots>
            {items.map((_, i) => (
                <PaginationDot key={`item:${i}`} isActive={i === index} />
            ))}
        </PaginationDots>
    );
}
```
