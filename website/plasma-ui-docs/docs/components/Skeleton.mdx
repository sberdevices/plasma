## На примере карточки

```tsx live
import React from 'react';
import {
    Card,
    CardBody,
    CardContent,
    LineSkeleton,
    RectSkeleton,
    Button,
    withSkeleton,
} from '@sberdevices/plasma-ui';

export function App() {
    const ButtonSkeleton = withSkeleton(Button);

    return (
        <Card style={{ width: '20rem' }} tabIndex={0}>
            <CardBody>
                <RectSkeleton width="100%" height="12rem" roundness={0} />
                <CardContent>
                    <LineSkeleton size="headline3" roundness={0} />
                    <LineSkeleton size="headline1" roundness={0} />
                    <LineSkeleton size="footnote1" roundness={0} />
                    <ButtonSkeleton
                        fullWidth
                        text="Загрузка..."
                        view="primary"
                        size="s"
                        scaleOnInteraction={false}
                        outlined={false}
                        skeleton={true}
                        style={{ marginTop: '1rem' }}
                        tabIndex={-1}
                    />
                </CardContent>
            </CardBody>
        </Card>
    );
}
```
